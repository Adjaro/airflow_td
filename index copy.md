    def extract_usa():
        return {"ventes": [500, 600, 700]}

    # Configuration du DAG

        catchup=False,
    )

        task_id='extract_france',
        python_callable=extract_france,
        dag=dag,

        task_id='extract_usa',
        python_callable=extract_usa,
        dag=dag,
